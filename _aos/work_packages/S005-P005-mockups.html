# scrapers/facebook_manual.py class ManualFacebookScraper(BaseScraper): """Facebook manual listing scraper with LLM-based Hebrew extraction.""" def __init__(self, source_id: str, search_url: str, city: str): super().__init__(source_id, search_url, city) self.input_path = Path(search_url) # search_url = file path def fetch_listings(self) -> list[ScrapedListing]: # 1. Validate LLM config if not self._check_llm_config(): return [] # 2. Read + validate input file posts = self._read_input_file() if not posts: return [] # 3. Parse each post via LLM listings = [] for post in posts: parsed = parse_rental_post(post["text"], post.get("group_name", "")) if parsed and parsed["is_rental_offer"]: listing = self._to_scraped_listing(post, parsed) listings.append(listing) # 4. Deduplicate return self._deduplicate(listings)

Facebook Email Notification Formats

Three notification types to parse

Single Post

Subject: "New post in [Group]"
  • 1 post per email
  • Full snippet (200-300 chars)
  • Direct post URL
  • Author display name

Digest

Subject: "N new posts in [Group]"
  • Multiple posts per email
  • Shorter snippets per post
  • Individual post URLs
  • Grouped by time period

Popular Posts

Subject: "Popular posts in [Group]"
  • Algorithmic selection
  • May include older posts
  • Higher engagement posts
  • Engagement metrics shown

Deduplication Pipeline (Ordered)

4-layer deterministic dedup (evaluated in order)
Incoming listing from email
↓
P1: Email message-ID
Already processed? → skip entire email
↓ not seen
P2: Post URL exact match
Same raw_url as existing listing? → skip
↓ no match
P3: Text hash (within batch)
SHA-256 of normalized text → first wins
↓ unique
P4: Fuzzy listing match
city + price ±10% + rooms → cross_listed
↓ pass
Add to results
